					draw,

				draw,
				fill=white,

				minimum width=\R,

			];

			line width=0.2mm,
			dotted,
			color=black,
			>=stealth'
		];

		\node [cylinder, aspect=.5, minimum width=2*\R, minimum height=\R] (data) [xshift=2cm, yshift=-\R] {} ;
		\node [cylinder, aspect=.5, style=database, minimum width=2*\R, minimum height=\R] (data) [xshift=2cm, yshift=-0.7cm] {} ;
		
		\path [<->, style=arrow] (0,0) edge [bend right] node [style=doc] {} ($(data)$);
		
		\node (text) [above of=data] {\resizebox{1.3cm}{!}{\textsc{dlvhex}}};

		\setlength{\textyshift}{-1.5mm}
		\ifshowgroup
			\node (text2) [above of=data, yshift=\the\textyshift] {\resizebox{1.3cm}{!}{\textsc{Knowledge-based Systems Group}}};
			\addtolength{\textyshift}{-0.7mm}
		\fi
		\ifshowinstitute
			\node (text2) [above of=data, yshift=\the\textyshift] {\resizebox{1.3cm}{!}{\textsc{Institute of Information Systems}}};

		\fi
